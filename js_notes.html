Свойства и методи на масивите

var arr = ['Нар', 'Ягода', 'Круша', 'Ябълка']

let a = arr[0] - присвоява на a стойността на първия елемент на масива т.е.'Нар'

arr[1] = 'Малина' - присвоява на втория елемент на масива стойността 'Малина'

arr.length - връща 4 - дължината на масива (т.е броят на елементите на масива)

arr[arr.length - 1] - връща 'Ябълка' т.е. стойността на последния елемент на масива

arr.pop() - премахва последния елемент на масива

let b = arr.pop() - премахва последния елемент от масива и присвоява стойността му на b

arr.push('Киви') - добавя елемент със стойност 'Киви' в края на масива

arr.shift() - премахва първия елемент на масива

let c = arr.shift() - премахва първия елемент от масива и присвоява стойността му на c

arr.unshift('Касис') - добавя елемент със стойност 'Касис' в началото на масива

Обхождащи методи на масивите чрез стрелкови функции

let brr = [2, 8, 1, 5, 23, 18];

let crr = ['1', '-10', '15', '20', '-5', '8', '14'];

brr.map(element => element *= 2)
- връща [4, 16, 2, 10, 46, 36] т.е копие на масива с удвоени стойности

brr.map((element, index) => element + index)
- връща [2, 9, 3, 8, 27, 23] копие на масива с елементи, към които е добавен индекса

crr.map(element => parseInt(element))
- връща копие на crr с елементи от числов тип

brr.forEach(element => console.log(element))
- извежда елементите на масива в конзолата на отделни редове

brr.forEach((element, i) => console.log(i))
- на отделни редове в конзолата извежда стойността на итератора при всяка стъпка

brr.filter(element => element % 2 == 0)
- връща [2, 8, 18] т.е. копие на масива с елементи, които отговарят на условието

brr.every(element => element % 2 == 0)
- връща true, ако всеки елелемент на brr отговаря на условието или brr e празен масив

brr.some(element => element > 0)
- връща false, ако нито един елелемент на brr не отговаря на условието или brr е празен

brr.reduce((acc, element) => acc + element)
- връща 57 т.е. сумата от елементите на масива

brr.reduce((acc, element) => acc + element, 100)
- връща 157 т.е. сумата от елементите на масива + 100 - началната стойност на acc

аrr.reduceRight((acc, element) => acc + element, 100)
- като reduce, но обработва масива от дясно на ляво. Връща 100ЯбълкаКрушаЯгодаНар
